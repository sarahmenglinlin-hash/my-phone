## 4. 关键算法与技术细节

### 4.1 RAG 语义检索算法

#### 目标
用户输入消息后，自动匹配相关的世界书条目，注入到 LLM 提示词中。

#### 实现步骤

**Step 1: 关键词提取**
```javascript
// 方案 A: 简单关键词提取 (初版)
KeywordExtractor.extract("我今天在学校上课，感到很开心")
  → ["学校", "上课", "开心"]

// 方案 B: 基于 TF-IDF 或 NLP 库 (后续优化)
使用 natural.js 或 compromise.js 库
```

**Step 2: 语义相似度计算**
```javascript
// 方案 A: 简单粗暴 - 词向量余弦相似度
// 使用预训练的中文词向量 (word2vec / fastText)
// 库: ml.js, numpy-js

similarity("上课", "学校") = 0.75
similarity("开心", "快乐") = 0.92

// 方案 B: Embedding API
// 调用 OpenAI / Cohere 的 embedding API
// 成本: 更高，但精度更好
```

**Step 3: 条目排序与注入**
```javascript
RAGService.retrieveRelevant(charId, keywords) {
  // 1. 获取该角色的所有世界书
  const worldbooks = WorldbookStore.getByCharId(charId)
  
  // 2. 计算每个条目与用户输入的相似度
  const scored = []
  for (let entry of allEntries) {
    const baseScore = calculateSimilarity(keywords, entry.keywords)
    const priorityBoost = entry.priority / 100  // 0-1 范围
    const finalScore = baseScore * 0.7 + priorityBoost * 0.3
    scored.push({ entry, score: finalScore })
  }
  
  // 3. 按相似度 + 优先级排序，得分 > 阈值的才注入
  const relevant = scored
    .filter(s => s.score > 0.3)  // 相似度阈值
    .sort((a, b) => b.score - a.score)
    .map(s => s.entry)
  
  // 4. 返回相关条目（无大小限制）
  return relevant
}
```

**Step 4: 提示词构建**
```javascript
PromptService.buildSystemPrompt(character, relevantEntries, preset) {
  const systemPrompt = `
你是 ${character.name}。

基本设定:
${character.personality}

相关背景信息:
${relevantEntries.map((e, i) => 
  `[${i+1}] ${e.title} (类别: ${e.category})\n${e.content}`
).join('\n\n')}

思考框架:
${preset.content}

请在回复前思考，确保符合你的性格设定和上述背景信息。
  `.trim()
  
  return systemPrompt
}
```

### 4.2 人物OOC防控

**策略**:
1. **提示词嵌入** (必须)
   - 完整人物设定在每次请求的系统消息里
   - 明确告诉模型："你是 [名字]，性格是 [具体描述]"

2. **CoT 预设** (推荐)
   - 用户可配置思维链，例如：
     ```
     1. 我是 [角色名]，我的性格是 [具体特点]
     2. 用户说的话意思是 ...
     3. 根据我的性格设定，我应该这样回复 ...
     ```
   - 模型会按照这个思考链进行回复，减少OOC

3. **世界书背景** (补充)
   - 提供背景故事细节，让模型更了解角色
   - 例如：学校、家庭、朋友圈等背景信息

4. **温度调整** (微调)
   - 用户可调整 temperature 参数
   - 低温度 (0.3-0.7): 更一致，更难OOC
   - 高温度 (1.2+): 更创意，可能OOC

## 5. IndexedDB 操作指南

### 5.1 初始化数据库

```javascript
// dbService.js

const DB_NAME = 'MiniPhoneDB'
const DB_VERSION = 1

const stores = {
  characters: {
    keyPath: 'id',
    indexes: [{ name: 'createdAt', keyPath: 'createdAt' }]
  },
  worldbooks: {
    keyPath: 'id',
    indexes: [{ name: 'createdAt', keyPath: 'createdAt' }]
  },
  conversations: {
    keyPath: 'id',
    indexes: [
      { name: 'charId', keyPath: 'charId' },
      { name: 'createdAt', keyPath: 'createdAt' }
    ]
  },
  presets: {
    keyPath: 'id',
    indexes: [{ name: 'isActive', keyPath: 'isActive' }]
  }
}

export class DBService {
  static async initDB() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(DB_NAME, DB_VERSION)
      
      request.onerror = () => reject(request.error)
      request.onsuccess = () => resolve(request.result)
      
      request.onupgradeneeded = (event) => {
        const db = event.target.result
        for (const [storeName, config] of Object.entries(stores)) {
          if (!db.objectStoreNames.contains(storeName)) {
            const store = db.createObjectStore(storeName, { keyPath: config.keyPath })
            config.indexes?.forEach(idx => {
              store.createIndex(idx.name, idx.keyPath)
            })
          }
        }
      }
    })
  }
}
```

### 5.2 常用操作

```javascript
// 增
async addCharacter(char) {
  const db = await this.initDB()
  const tx = db.transaction(['characters'], 'readwrite')
  const store = tx.objectStore('characters')
  return new Promise((resolve, reject) => {
    const request = store.add({ ...char, id: Date.now().toString() })
    request.onsuccess = () => resolve(request.result)
    request.onerror = () => reject(request.error)
  })
}

// 查
async getCharacterById(charId) {
  const db = await this.initDB()
  const tx = db.transaction(['characters'], 'readonly')
  const store = tx.objectStore('characters')
  return new Promise((resolve, reject) => {
    const request = store.get(charId)
    request.onsuccess = () => resolve(request.result)
    request.onerror = () => reject(request.error)
  })
}

// 改
async updateCharacter(charId, updates) {
  const db = await this.initDB()
  const tx = db.transaction(['characters'], 'readwrite')
  const store = tx.objectStore('characters')
  const char = await this.getCharacterById(charId)
  return new Promise((resolve, reject) => {
    const request = store.put({ ...char, ...updates, updatedAt: Date.now() })
    request.onsuccess = () => resolve(request.result)
    request.onerror = () => reject(request.error)
  })
}

// 删
async deleteCharacter(charId) {
  const db = await this.initDB()
  const tx = db.transaction(['characters'], 'readwrite')
  const store = tx.objectStore('characters')
  return new Promise((resolve, reject) => {
    const request = store.delete(charId)
    request.onsuccess = () => resolve()
    request.onerror = () => reject(request.error)
  })
}
```

### 6.2 统一的 API 调用接口

```javascript
// llmService.js

export class LLMService {
  
  /**
   * 流式聊天接口 (统一适配多个服务商)
   */
  static async streamChat(config, messages, systemPrompt) {
    // config = { provider, baseURL, apiKey, model, temperature, ... }
    
    // 1. 根据服务商适配请求格式
    const payload = this.adaptPayload(config, messages, systemPrompt)
    
    // 2. 发起 HTTP 请求
    const response = await fetch(
      `${config.baseURL}/v1/chat/completions`,
      {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${config.apiKey}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify(payload)
      }
    )
    
    if (!response.ok) {
      throw new Error(`API Error: ${response.status}`)
    }
    
    // 3. 处理流式响应
    const reader = response.body.getReader()
    const decoder = new TextDecoder()
    let buffer = ''
    
    while (true) {
      const { done, value } = await reader.read()
      if (done) break
      
      buffer += decoder.decode(value, { stream: true })
      
      // 按行处理 SSE 格式
      const lines = buffer.split('\n')
      buffer = lines.pop() // 保留未完成的行
      
      for (const line of lines) {
        if (line.startsWith('data: ')) {
          const data = JSON.parse(line.slice(6))
          if (data.choices[0].delta.content) {
            yield data.choices[0].delta.content  // 逐字符流出
          }
        }
      }
    }
  }
  
  /**
   * 根据服务商适配请求格式
   */
  static adaptPayload(config, messages, systemPrompt) {
    const basePayload = {
      model: config.model,
      messages: [
        { role: 'system', content: systemPrompt },
        ...messages
      ],
      temperature: config.temperature ?? 1.0,
      stream: true
    }
    
    // 特定服务商的调整 (如有)
    if (config.provider === 'anthropic') {
      basePayload.max_tokens = 2048
    }
    
    return basePayload
  }
  
  /**
   * 获取服务商的可用模型列表
   */
  static async getAvailableModels(config) {
    const response = await fetch(
      `${config.baseURL}/v1/models`,
      {
        headers: {
          'Authorization': `Bearer ${config.apiKey}`
        }
      }
    )
    const data = await response.json()
    return data.data.map(m => m.id)
  }
  
  /**
   * 测试 API 连接
   */
  static async testConnection(config) {
    try {
      const models = await this.getAvailableModels(config)
      return { success: true, message: `连接成功，找到 ${models.length} 个模型` }
    } catch (error) {
      return { success: false, message: error.message }
    }
  }
}
```

```javascript
async function importData(file) {
  const content = await file.text()
  const data = JSON.parse(content)
  
  // 冲突处理
  const conflictResolution = await askUser('覆盖还是合并？')
  
  if (conflictResolution === 'overwrite') {
    // 清空现有数据，导入新数据
    await DBService.clearAll()
  }
  
  // 逐个导入
  for (const char of data.characters) {
    await DBService.addCharacter(char)
  }
  
  for (const wb of data.worldbooks) {
    await DBService.addWorldbook(wb)
  }
  
  // 保存 API 配置
  if (data.apiConfig) {
    LocalStorage.setItem('api_config', data.apiConfig)
  }
  
  // 重新加载 UI
  location.reload()
}
```

## 8. 性能优化建议

### 8.1 IndexedDB 优化
- 使用索引查询而不是全表扫描
- 定期清理过期数据
- 分页加载聊天记录（虚拟滚动）

### 8.2 LLM 流式处理
- 使用 ReadableStream 逐字符显示
- 取消长时间未响应的请求 (10s超时)

### 8.3 内存管理
- 聊天界面只加载最近500条消息
- 及时释放已完成的 Promise

### 8.4 UI 优化
- Vue 组件层级扁平化
- 使用 v-show 替代 v-if 频繁切换
- 虚拟列表优化长列表 (nuxt-virtual-scroller)
